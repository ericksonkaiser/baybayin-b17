function syllabify(word) { // B17 RULE: CV, V, or single C lang word = word.toLowerCase().replace(/[^a-z]/g, ''); // Handle double letters: coffee → cofi, quezon → keson word = word.replace(/cc/g, 'k').replace(/qu/g, 'k').replace(/c/g, 'k'); word = word.replace(/([a-z])\1+/g, '$1'); const vowels = 'aeiou'; const syllables = []; let i = 0; while (i < word.length) { // Rule 1: Consonant + Vowel = CV syllable if (i + 1 < word.length && vowels.includes(word[i + 1])) { syllables.push(word[i] + word[i + 1]); i += 2; } // Rule 2: Single vowel = V syllable else if (vowels.includes(word[i])) { syllables.push(word[i]); i++; } // Rule 3: Single consonant = C + virama later else { syllables.push(word[i]); i++; } } return syllables; }
